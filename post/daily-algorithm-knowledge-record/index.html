double myPow(double x, int n) {
    if (x == 0 || x == 1) return x;
    if (n == 0) return 1;
    long nBak = n;
    if (nBak < 0) {
        nBak = -nBak;
        x = 1 / x;
    }
    double ans = 1, weight = x;
    while (nBak != 0) {
        if ((nBak & 1) == 1) {
            ans *= weight;
        }
        weight *= weight;
        nBak = nBak >> 1;
    }
    return ans;
}

class Trie {
    private boolean isEnd;
    private Trie[] words;

    public Trie() {
        words = new Trie[26];
        isEnd = false;
    }

    public void insert(String word) {
        Trie cur = this;
        for (char c : word.toCharArray()) {
            if (cur.words[c - 'a'] == null) {
                cur.words[c - 'a'] = new Trie();
            }
            cur = cur.words[c - 'a'];
        }
        cur.isEnd = true;
    }

    public boolean search(String word) {
        Trie cur = this;
        for (char c : word.toCharArray()) {
            if (cur.words[c - 'a'] == null) return false;
            cur = cur.words[c - 'a'];
        }
        return cur.isEnd;
    }

    public boolean startsWith(String prefix) {
        Trie cur = this;
        for (char c : prefix.toCharArray()) {
            if (cur.words[c - 'a'] == null) return false;
            cur = cur.words[c - 'a'];
        }
        return true;
    }
}
