1. 十进制转二进制。代码实现

实现逻辑:

  1. 获取最后一位，判断是1还是0
  2. 向右移动一位

代码：

void tenToBinary(int n) {
    string ans;
    while (n > 0) {
        ans.push_back((n & 1) + '0');
        n = n >> 1;
    }
    reverse(ans.begin(), ans.end());
    cout << ans << endl;
}

2. 高效实现Pow(x,n)

可以通过快速幂来实现Pow(x,n)。时间复杂度 O(logn)

实现逻辑：

代码：

double myPow(double x, int n) {
    if (x == 0 || x == 1) return x;
    if (n == 0) return 1;
    long nBak = n;
    if (nBak < 0) {
        nBak = -nBak;
        x = 1 / x;
    }
    double ans = 1, weight = x;
    while (nBak != 0) {
        if ((nBak & 1) == 1) {
            ans *= weight;
        }
        weight *= weight;
        nBak = nBak >> 1;
    }
    return ans;
}

1. 优先级队列实现堆排序

Java可直接使用PriorityQueue这个类来实现堆排序。

代码

  private PriorityQueue<Integer> queue;
    private int size;

    public int[] getLeastNumbers(int[] arr, int k) {
        int[] ans = new int[k];
        this.queue = new PriorityQueue<>((o1, o2) -> o2 - o1);
        this.size = 0;

        if (k == 0)
            return ans;

        for(int element:arr){
            if(this.size==k){  // queue已经满了,如果堆顶元素>element,poll,
                int topElement = queue.poll();
                queue.add(Math.min(topElement, element));
            } else {
                this.queue.add(element);
                this.size++;
             }
         }

         int index=0;
         while(!queue.isEmpty()){
             ans[index++] = queue.poll();
         }
        return ans;
    }

优先级队列用途

  • 求n个数中前k小
  • 求n个数中前k大

2. Trie(前缀树) 实现

每个节点都包含26个数据部分和一个标识是否为字符串终止的标识。

public class Test {
    public static void main(String[] args) {
        Trie trie = new Trie();
        trie.insert("apple");
        trie.search("apple");
    }
}

class Trie {
    private boolean isEnd;
    private Trie[] words;

    public Trie() {
        words = new Trie[26];
        isEnd = false;
    }

    public void insert(String word) {
        Trie cur = this;
        for (char c : word.toCharArray()) {
            if (cur.words[c - 'a'] == null) {
                cur.words[c - 'a'] = new Trie();
            }
            cur = cur.words[c - 'a'];
        }
        cur.isEnd = true;
    }

    public boolean search(String word) {
        Trie cur = this;
        for (char c : word.toCharArray()) {
            if (cur.words[c - 'a'] == null) return false;
            cur = cur.words[c - 'a'];
        }
        return cur.isEnd;
    }

    public boolean startsWith(String prefix) {
        Trie cur = this;
        for (char c : prefix.toCharArray()) {
            if (cur.words[c - 'a'] == null) return false;
            cur = cur.words[c - 'a'];
        }
        return true;
    }
}
